import networkx as nx
import matplotlib.pyplot as plt

colors = ["red", "green", "blue", "yellow"]

def is_safe(node, color, assignment, graph):
    for neighbor in graph.neighbors(node):
        if neighbor in assignment and assignment[neighbor] == color:
            return False
    return True

def graph_coloring(graph, nodes, assignment, index):
    if index == len(nodes):
        return True
    
    node = nodes[index]
    
    for color in colors:
        if is_safe(node, color, assignment, graph):
            assignment[node] = color
            if graph_coloring(graph, nodes, assignment, index + 1):
                return True
            del assignment[node]
    
    return False

G = nx.Graph()

edges = [
    (1,2),(1,3),(2,3),(2,4),(3,5),
    (4,5),(4,6),(5,7),(6,7),(6,8),
    (7,9),(8,9),(9,10),(8,10)
]

G.add_edges_from(edges)

nodes = list(G.nodes())
assignment = {}

graph_coloring(G, nodes, assignment, 0)

node_colors = [assignment[node] for node in G.nodes()]

pos = nx.spring_layout(G)
nx.draw(G, pos, with_labels=True, node_color=node_colors, node_size=800)

plt.title("Graph Coloring using 4 Colors (Backtracking)")
plt.show()
